### Step 2: Create PDF Export Utility (1-2 hours)

Create `src/utils/pdfExport.js`:

```javascript
import jsPDF from 'jspdf';
import html2canvas from 'html2canvas';

/**
 * Export dashboard cards to PDF
 * @param {Object} options - Export options
 * @param {Array} options.selectedCards - Array of card IDs to export
 * @param {string} options.divisionName - Division name for title
 * @param {Object} options.companyCurrency - Currency object
 * @param {string} options.logoBase64 - Base64 encoded logo
 * @param {Function} options.onProgress - Progress callback (current, total)
 */
export const exportDashboardToPDF = async ({
  selectedCards,
  divisionName,
  companyCurrency,
  logoBase64,
  onProgress
}) => {
  // Initialize PDF
  const pdf = new jsPDF({
    orientation: 'portrait',
    unit: 'mm',
    format: 'a4',
    compress: true
  });
  
  const pageWidth = pdf.internal.pageSize.getWidth();
  const pageHeight = pdf.internal.pageSize.getHeight();
  const margin = 15;
  const contentWidth = pageWidth - (2 * margin);
  
  let currentY = margin;
  
  // Add header
  if (logoBase64) {
    try {
      pdf.addImage(logoBase64, 'PNG', margin, currentY, 30, 15);
    } catch (e) {
      console.warn('Could not add logo:', e);
    }
    currentY += 20;
  }
  
  // Add title
  pdf.setFontSize(20);
  pdf.setTextColor(26, 144, 255);
  pdf.text(`${divisionName} Dashboard Report`, pageWidth / 2, currentY, {
    align: 'center'
  });
  currentY += 10;
  
  // Add date
  pdf.setFontSize(10);
  pdf.setTextColor(100, 100, 100);
  const dateStr = new Date().toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  });
  pdf.text(`Generated: ${dateStr}`, pageWidth / 2, currentY, {
    align: 'center'
  });
  currentY += 15;
  
  // Process each card
  for (let i = 0; i < selectedCards.length; i++) {
    const cardId = selectedCards[i];
    
    // Update progress
    if (onProgress) {
      onProgress(i + 1, selectedCards.length);
    }
    
    // Get element
    const element = document.getElementById(cardId);
    if (!element) {
      console.warn(`Card element not found: ${cardId}`);
      continue;
    }
    
    // Add section title
    const cardTitle = getCardTitle(cardId);
    pdf.setFontSize(14);
    pdf.setTextColor(26, 144, 255);
    
    // Check if title fits on current page
    if (currentY + 8 > pageHeight - margin) {
      pdf.addPage();
      currentY = margin;
    }
    
    pdf.text(cardTitle, margin, currentY);
    currentY += 8;
    
    try {
      // Capture element as image
      const canvas = await html2canvas(element, {
        scale: 2, // High resolution
        logging: false,
        useCORS: true,
        backgroundColor: '#ffffff',
        windowWidth: 1200,
        onclone: (clonedDoc) => {
          // Hide any buttons or interactive elements in the clone
          const clonedElement = clonedDoc.getElementById(cardId);
          if (clonedElement) {
            const buttons = clonedElement.querySelectorAll('button, .ant-btn');
            buttons.forEach(btn => btn.style.display = 'none');
          }
        }
      });
      
      const imgData = canvas.toDataURL('image/png');
      const imgHeight = (canvas.height * contentWidth) / canvas.width;
      
      // Check if image fits on current page
      if (currentY + imgHeight > pageHeight - margin) {
        pdf.addPage();
        currentY = margin;
      }
      
      // Add image to PDF
      pdf.addImage(imgData, 'PNG', margin, currentY, contentWidth, imgHeight);
      currentY += imgHeight + 10;
      
      // Add page break after major sections
      if (shouldAddPageBreak(cardId) && i < selectedCards.length - 1) {
        pdf.addPage();
        currentY = margin;
      }
      
    } catch (error) {
      console.error(`Error capturing card ${cardId}:`, error);
      
      // Add error message to PDF
      pdf.setFontSize(10);
      pdf.setTextColor(220, 53, 69);
      pdf.text(`Error capturing ${cardTitle}`, margin, currentY);
      currentY += 10;
    }
  }
  
  // Add footer to all pages
  const totalPages = pdf.internal.getNumberOfPages();
  for (let i = 1; i <= totalPages; i++) {
    pdf.setPage(i);
    pdf.setFontSize(8);
    pdf.setTextColor(150, 150, 150);
    
    // Page number
    pdf.text(
      `Page ${i} of ${totalPages}`,
      pageWidth / 2,
      pageHeight - 10,
      { align: 'center' }
    );
    
    // Company info
    pdf.text(
      'Generated by ProPackHub PEBI',
      margin,
      pageHeight - 10
    );
  }
  
  // Save the PDF
  const timestamp = new Date().toISOString().slice(0, 10);
  const filename = `${divisionName}_Dashboard_${timestamp}.pdf`;
  pdf.save(filename);
  
  return filename;
};

/**
 * Get display title for a card
 */
const getCardTitle = (cardId) => {
  const titles = {
    'divisionalKpis': '📊 Divisional KPIs',
    'profitLoss': '💰 Profit & Loss Statement',
    'salesVolumeChart': '📈 Sales & Volume Analysis',
    'marginAnalysis': '📉 Margin Analysis',
    'manufacturingCost': '🏭 Manufacturing Cost',
    'belowGPExpenses': '💸 Below GP Expenses',
    'combinedTrends': '📊 Combined Trends',
    'productGroups': '📦 Product Groups',
    'salesBySalesReps': '🧑‍💼 Sales by Sales Reps',
    'salesByCustomers': '👥 Sales by Customers',
    'salesByCountries': '🌍 Sales by Countries',
    'budgetVsActual': '🎯 Budget vs Actual Bridge'
  };
  return titles[cardId] || cardId;
};

/**
 * Determine if a page break should be added after this card
 */
const shouldAddPageBreak = (cardId) => {
  // Add page breaks after major sections for better readability
  const pageBreakAfter = [
    'divisionalKpis',
    'profitLoss',
    'combinedTrends'
  ];
  return pageBreakAfter.includes(cardId);
};
```

```javascript
// Helper to convert logo to base64
const getBase64Logo = async () => {
  try {
    const response = await fetch(ipTransparentLogo);
    const blob = await response.blob();
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onloadend = () => resolve(reader.result);
      reader.onerror = reject;
      reader.readAsDataURL(blob);
    });
  } catch (error) {
    console.warn('Could not load logo:', error);
    return null;
  }
};
```

## Troubleshooting

### Issue: Charts not captured correctly

**Solution:** Ensure charts are fully rendered before capture

```javascript
// Wait for charts to render
await new Promise(resolve => setTimeout(resolve, 1000));

// Then capture
const canvas = await html2canvas(element, { /* options */ });
```

### Issue: File size too large

**Solutions:**
1. Reduce scale: `scale: 1.5` instead of `2`
2. Use JPEG: `canvas.toDataURL('image/jpeg', 0.85)`
3. Compress PDF: `compress: true` in jsPDF options

### Issue: Page breaks in wrong places

**Solution:** Adjust page break logic

```javascript
// More aggressive page breaks
if (currentY + imgHeight > pageHeight - margin - 50) { // Add 50mm buffer
  pdf.addPage();
  currentY = margin;
}
```

### Issue: Fonts not rendering correctly

**Solution:** Embed custom fonts

```javascript
import { jsPDF } from 'jspdf';
import customFont from './fonts/CustomFont-Regular.ttf';

// Add font to jsPDF
pdf.addFileToVFS('CustomFont-Regular.ttf', customFont);
pdf.addFont('CustomFont-Regular.ttf', 'CustomFont', 'normal');
pdf.setFont('CustomFont');
```
